/**
* 获取传入月份所属季度的最后一一个月份
*
* @param dataMonth
* @return
* @author CHENXIAOTI496 2020年7月8日
*/
public static final String getQuarterLastMonth(String dataMonth) {
int month = Integer.parseInt(dataMonth.substring(4, 6));
int minQuarter = (month - 1) / 3 + 1;
int lastMonth = (minQuarter - 1) * 3 + 3;
return dataMonth.substring(0, 4) + (Integer.toString(lastMonth).length() < 2 ? "0" : "" ) + lastMonth; } /** *
	获取传入季度所属季度的最后一个月份 * * @param dataQuater * @return * @author CHENXIAOTI496 2020年7月8日 */ public static final String
	getQuarterLastMonthByQuarter(String dataQuater) { String quater=dataQuater.substring(4, 6); if ("Q1".equals(quater))
	{ return dataQuater.substring(0, 4) + "03" ; } else if ("Q2".equals(quater)) { return dataQuater.substring(0, 4)
	+ "06" ; } else if ("Q3".equals(quater)) { return dataQuater.substring(0, 4) + "09" ; } else { return
	dataQuater.substring(0, 4) + "12" ; } }
